```python
class ADEvidenceLog(models.Model):
    EVENT_TYPES = [
        ('WORKFLOW_STARTED', 'Workflow Started'),
        ('WORKFLOW_COMPLETED', 'Workflow Completed'),
        ('WORKFLOW_FAILED', 'Workflow Failed'),
        ('PHASE_COMPLETED', 'Phase Completed'),
        ('FINDING_DISCOVERED', 'Finding Discovered'),
        ('EXPOSURE_DISCOVERED', 'Exposure Discovered'),
        ('DATA_INGESTED', 'Data Ingested'),
        ('GRAPH_SNAPSHOT', 'Graph Snapshot'),
        ('REPORT_GENERATED', 'Report Generated'),
        ('ANALYST_ACTION', 'Analyst Action'),
    ]
    assessment = models.ForeignKey(
        ADAssessment, on_delete=models.CASCADE, related_name='evidence_log')
    event_type = models.CharField(max_length=50, choices=EVENT_TYPES)
    actor = models.CharField(max_length=255, default='system',
                             help_text="'system' for automated events, username for analyst")
    details = models.JSONField(default=dict)
    timestamp = models.DateTimeField(default=timezone.now)

    class Meta:
        db_table = 'plugin_ad_evidence_log'
        ordering = ['-timestamp']

    def __str__(self):
        return f"[{self.event_type}] {self.assessment} @ {self.timestamp:%Y-%m-%d %H:%M}"
```

```python
# r3ngine-plugins/active_directory/backend/reporting/engine.py
import logging
from datetime import timezone as tz
from typing import Any, Dict

logger = logging.getLogger(__name__)


class ReportingEngine:
    """
    Compiles an ADAssessment into a structured report dict consumed by
    PDFRenderer and JSONRenderer.
    """

    @classmethod
    def compile(cls, assessment_id: int) -> Dict[str, Any]:
        from ..models import (ADAssessment, ADDomain, ADTrust, ADExposure,
                               ADFinding, ADEvidenceLog)

        assessment = ADAssessment.objects.get(pk=assessment_id)
        domains = list(ADDomain.objects.filter(assessment=assessment)
                       .values('fqdn', 'name', 'forest_root', 'functional_level',
                               'dc_count', 'user_count', 'group_count', 'computer_count'))
        trusts = list(ADTrust.objects.filter(assessment=assessment)
                      .select_related('source_domain')
                      .values('source_domain__fqdn', 'target_domain_name',
                              'direction', 'trust_type', 'is_transitive',
                              'is_selective_auth', 'risk_score'))
        exposures = list(ADExposure.objects.filter(assessment=assessment)
                         .select_related('correlated_domain')
                         .values('hostname', 'ip_address', 'port', 'exposure_type',
                                 'correlated_domain__fqdn', 'risk_score'))
        findings = list(ADFinding.objects.filter(assessment=assessment)
                        .values('title', 'description', 'severity', 'status',
                                'finding_type', 'affected_object', 'remediation'))
        timeline = list(ADEvidenceLog.objects.filter(assessment=assessment)
                        .order_by('timestamp')
                        .values('event_type', 'actor', 'details', 'timestamp'))

        # Serialize timestamps
        for entry in timeline:
            if entry.get('timestamp'):
                entry['timestamp'] = entry['timestamp'].isoformat()

        critical_findings = sum(1 for f in findings if f['severity'] == 'CRITICAL')
        high_exposures = sum(1 for e in exposures if e['risk_score'] >= 70.0)

        return {
            'metadata': {
                'assessment_name': assessment.name,
                'target_domain': assessment.target_domain,
                'status': assessment.status,
                'created_at': assessment.created_at.isoformat(),
                'started_at': assessment.started_at.isoformat() if assessment.started_at else None,
                'completed_at': assessment.completed_at.isoformat() if assessment.completed_at else None,
                'generated_at': __import__('datetime').datetime.now(tz.utc).isoformat(),
            },
            'executive_summary': {
                'total_domains': len(domains),
                'total_findings': len(findings),
                'total_exposures': len(exposures),
                'critical_findings': critical_findings,
                'high_risk_exposures': high_exposures,
                'trust_count': len(trusts),
            },
            'domain_inventory': domains,
            'trust_analysis': trusts,
            'exposure_analysis': exposures,
            'findings': findings,
            'timeline': timeline,
        }
```

```python
# r3ngine-plugins/active_directory/backend/reporting/pdf_renderer.py
"""
PDF report renderer using ReportLab.

Produces an enterprise-formatted PDF with:
- Cover page (assessment name, target, date, status)
- Executive summary table
- Domain inventory
- Trust analysis table
- Exposure dashboard
- Findings list
- Assessment timeline
"""
import io
import logging
from typing import Any, Dict

logger = logging.getLogger(__name__)

try:
    from reportlab.lib import colors
    from reportlab.lib.pagesizes import A4
    from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
    from reportlab.lib.units import cm
    from reportlab.platypus import (
        SimpleDocTemplate, Paragraph, Spacer, Table, TableStyle,
        HRFlowable, PageBreak
    )
    REPORTLAB_AVAILABLE = True
except ImportError:
    REPORTLAB_AVAILABLE = False
    logger.warning("ReportLab not installed — PDF reports unavailable. "
                   "Install with: pip install reportlab")


BRAND_DARK = colors.HexColor('#0a0a0f')
BRAND_PRIMARY = colors.HexColor('#00f3ff')
BRAND_DANGER = colors.HexColor('#ff003c')
BRAND_WARN = colors.HexColor('#ff6b35')
BRAND_SAFE = colors.HexColor('#4ecdc4')
BRAND_TEXT = colors.HexColor('#e0e0e0')
BRAND_SUBTLE = colors.HexColor('#888888')

SEVERITY_COLORS = {
    'CRITICAL': BRAND_DANGER,
    'HIGH': BRAND_WARN,
    'MEDIUM': colors.HexColor('#f7c59f'),
    'LOW': BRAND_SAFE,
    'INFO': BRAND_SUBTLE,
}


class PDFRenderer:

    @classmethod
    def render(cls, report: Dict[str, Any]) -> bytes:
        if not REPORTLAB_AVAILABLE:
            raise RuntimeError(
                "ReportLab is required for PDF generation. "
                "Install with: pip install reportlab")

        buf = io.BytesIO()
        doc = SimpleDocTemplate(buf, pagesize=A4,
                                 leftMargin=2 * cm, rightMargin=2 * cm,
                                 topMargin=2 * cm, bottomMargin=2 * cm)
        styles = getSampleStyleSheet()
        story = []

        # Styles
        h1 = ParagraphStyle('H1', parent=styles['Heading1'],
                             textColor=BRAND_PRIMARY, fontSize=20,
                             fontName='Helvetica-Bold', spaceAfter=6)
        h2 = ParagraphStyle('H2', parent=styles['Heading2'],
                             textColor=BRAND_TEXT, fontSize=13,
                             fontName='Helvetica-Bold', spaceAfter=4)
        body = ParagraphStyle('Body', parent=styles['Normal'],
                               textColor=BRAND_TEXT, fontSize=9, spaceAfter=3)
        meta = ParagraphStyle('Meta', parent=styles['Normal'],
                               textColor=BRAND_SUBTLE, fontSize=8)

        meta_data = report.get('metadata', {})
        exec_summary = report.get('executive_summary', {})

        # --- Cover page ---
        story.append(Spacer(1, 3 * cm))
        story.append(Paragraph("ACTIVE DIRECTORY INTELLIGENCE REPORT", h1))
        story.append(HRFlowable(width="100%", thickness=1, color=BRAND_PRIMARY))
        story.append(Spacer(1, 0.5 * cm))

        cover_data = [
            ['Assessment', meta_data.get('assessment_name', '')],
            ['Target Domain', meta_data.get('target_domain', '')],
            ['Status', meta_data.get('status', '')],
            ['Started', meta_data.get('started_at', 'N/A')],
            ['Completed', meta_data.get('completed_at', 'N/A')],
            ['Generated', meta_data.get('generated_at', '')],
        ]
        cover_table = Table(cover_data, colWidths=[4 * cm, 13 * cm])
        cover_table.setStyle(TableStyle([
            ('TEXTCOLOR', (0, 0), (-1, -1), BRAND_TEXT),
            ('FONTSIZE', (0, 0), (-1, -1), 9),
            ('FONTNAME', (0, 0), (0, -1), 'Helvetica-Bold'),
            ('BOTTOMPADDING', (0, 0), (-1, -1), 6),
        ]))
        story.append(cover_table)
        story.append(PageBreak())

        # --- Executive summary ---
        story.append(Paragraph("EXECUTIVE SUMMARY", h2))
        story.append(HRFlowable(width="100%", thickness=0.5, color=BRAND_SUBTLE))
        story.append(Spacer(1, 0.3 * cm))

        exec_data = [
            ['Metric', 'Value'],
            ['Domains Discovered', str(exec_summary.get('total_domains', 0))],
            ['Total Findings', str(exec_summary.get('total_findings', 0))],
            ['Critical Findings', str(exec_summary.get('critical_findings', 0))],
            ['Total Exposures', str(exec_summary.get('total_exposures', 0))],
            ['High-Risk Exposures', str(exec_summary.get('high_risk_exposures', 0))],
            ['Trust Relationships', str(exec_summary.get('trust_count', 0))],
        ]
        exec_table = Table(exec_data, colWidths=[8 * cm, 9 * cm])
        exec_table.setStyle(TableStyle([
            ('BACKGROUND', (0, 0), (-1, 0), BRAND_PRIMARY),
            ('TEXTCOLOR', (0, 0), (-1, 0), BRAND_DARK),
            ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
            ('TEXTCOLOR', (0, 1), (-1, -1), BRAND_TEXT),
            ('FONTSIZE', (0, 0), (-1, -1), 9),
            ('ROWBACKGROUNDS', (0, 1), (-1, -1),
             [colors.HexColor('#141420'), colors.HexColor('#0a0a0f')]),
            ('GRID', (0, 0), (-1, -1), 0.5, BRAND_SUBTLE),
            ('BOTTOMPADDING', (0, 0), (-1, -1), 5),
            ('TOPPADDING', (0, 0), (-1, -1), 5),
        ]))
        story.append(exec_table)
        story.append(Spacer(1, 0.5 * cm))

        # --- Findings ---
        findings = report.get('findings', [])
        if findings:
            story.append(PageBreak())
            story.append(Paragraph("FINDINGS", h2))
            story.append(HRFlowable(width="100%", thickness=0.5, color=BRAND_SUBTLE))
            story.append(Spacer(1, 0.3 * cm))

            for finding in findings:
                sev = finding.get('severity', 'INFO')
                sev_color = SEVERITY_COLORS.get(sev, BRAND_SUBTLE)
                story.append(Paragraph(
                    f'<font color="#{sev_color.hexval()[1:]}">[{sev}]</font> '
                    f'{finding.get("title", "")}',
                    body))
                if finding.get('description'):
                    story.append(Paragraph(finding['description'], meta))
                if finding.get('affected_object'):
                    story.append(Paragraph(
                        f'Affected: {finding["affected_object"]}', meta))
                if finding.get('remediation'):
                    story.append(Paragraph(
                        f'Remediation: {finding["remediation"]}', meta))
                story.append(Spacer(1, 0.2 * cm))

        # --- Exposure analysis ---
        exposures = report.get('exposure_analysis', [])
        if exposures:
            story.append(PageBreak())
            story.append(Paragraph("EXPOSURE ANALYSIS", h2))
            story.append(HRFlowable(width="100%", thickness=0.5, color=BRAND_SUBTLE))
            story.append(Spacer(1, 0.3 * cm))

            exp_header = ['Hostname', 'Type', 'Risk', 'Correlated Domain']
            exp_data = [exp_header] + [
                [
                    e.get('hostname', '')[:40],
                    e.get('exposure_type', ''),
                    f"{e.get('risk_score', 0):.0f}",
                    e.get('correlated_domain__fqdn') or '—',
                ]
                for e in sorted(exposures, key=lambda x: -x.get('risk_score', 0))
            ]
            exp_table = Table(exp_data, colWidths=[6 * cm, 3 * cm, 2 * cm, 6 * cm])
            exp_table.setStyle(TableStyle([
                ('BACKGROUND', (0, 0), (-1, 0), BRAND_PRIMARY),
                ('TEXTCOLOR', (0, 0), (-1, 0), BRAND_DARK),
                ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
                ('TEXTCOLOR', (0, 1), (-1, -1), BRAND_TEXT),
                ('FONTSIZE', (0, 0), (-1, -1), 8),
                ('GRID', (0, 0), (-1, -1), 0.3, BRAND_SUBTLE),
                ('ROWBACKGROUNDS', (0, 1), (-1, -1),
                 [colors.HexColor('#141420'), colors.HexColor('#0a0a0f')]),
                ('BOTTOMPADDING', (0, 0), (-1, -1), 4),
                ('TOPPADDING', (0, 0), (-1, -1), 4),
            ]))
            story.append(exp_table)

        doc.build(story)
        buf.seek(0)
        return buf.read()
```

```python
def run_ad_reporting_activity(params: dict) -> dict:
    """Generate PDF and JSON reports for the assessment."""
    assessment_id = params['assessment_id']

    _send_ws_update(assessment_id, 'phase_started', {

    })

    try:
        from .reporting.engine import ReportingEngine
        from .reporting.json_renderer import JSONRenderer

        report = ReportingEngine.compile(assessment_id)
        json_output = JSONRenderer.render(report)

        # Store JSON snapshot as evidence
        from .models import ADGraphSnapshot
        ADGraphSnapshot.objects.create(
            assessment_id=assessment_id,
            snapshot_type='assessment_report_json',
            graph_data={'report_json': json_output[:50000]},
        )

        _log_evidence(assessment_id, 'REPORT_GENERATED', {
            'format': 'json',
            'finding_count': len(report.get('findings', [])),
            'exposure_count': len(report.get('exposure_analysis', [])),
        })
    except Exception as exc:
        logger.error(f"[AD Reporting] Report generation failed: {exc}")
        _send_ws_update(assessment_id, 'phase_error', {
            'phase': 'reporting',
            'error': str(exc),
        })
        return {'status': 'error', 'error': str(exc)}

    _send_ws_update(assessment_id, 'phase_completed', {
        'phase': 'reporting',
        'message': 'Reports generated successfully',
    })

    return {'status': 'completed'}
```

```python
    @action(detail=True, methods=['get'], url_path='report')
    def report(self, request, pk=None):
        """
        Generate and return an assessment report.
        ?format=pdf  — returns PDF file download
        ?format=json — returns JSON report
        """
        assessment = self.get_object()
        if assessment.status not in ('COMPLETED', 'RUNNING'):
            return Response(
                {'error': 'Report only available for completed or running assessments.'},
                status=status.HTTP_400_BAD_REQUEST,
            )

        fmt = request.query_params.get('format', 'json').lower()

        try:
            from .reporting.engine import ReportingEngine
            report = ReportingEngine.compile(assessment.id)

            if fmt == 'pdf':
                from .reporting.pdf_renderer import PDFRenderer
                from django.http import HttpResponse
                pdf_bytes = PDFRenderer.render(report)
                response = HttpResponse(pdf_bytes, content_type='application/pdf')
                safe_name = assessment.name.replace(' ', '_')[:50]
                response['Content-Disposition'] = (
                    f'attachment; filename="AD_Report_{safe_name}.pdf"')
                return response
            else:
                from .reporting.json_renderer import JSONRenderer
                import json
                from django.http import HttpResponse
                json_str = JSONRenderer.render(report)
                response = HttpResponse(json_str, content_type='application/json')
                safe_name = assessment.name.replace(' ', '_')[:50]
                response['Content-Disposition'] = (
                    f'attachment; filename="AD_Report_{safe_name}.json"')
                return response

        except Exception as exc:
            logger.error(f"[AD Report] Generation failed: {exc}")
            return Response({'error': str(exc)},
                            status=status.HTTP_500_INTERNAL_SERVER_ERROR)

    @action(detail=True, methods=['get'], url_path='evidence')
    def evidence(self, request, pk=None):
        """Return the assessment evidence log (timeline)."""
        assessment = self.get_object()
        from .models import ADEvidenceLog
        logs = ADEvidenceLog.objects.filter(assessment=assessment)
        data = [
            {
                'event_type': log.event_type,
                'actor': log.actor,
                'details': log.details,
                'timestamp': log.timestamp.isoformat(),
            }
            for log in logs
        ]
        return Response(data)
```

```tsx
import React, { useState } from 'react';
import { useParams } from '@tanstack/react-router';
import {
  Box, Typography, Paper, Button, List, ListItem,
  ListItemText, ListItemIcon, Divider, Alert, CircularProgress
} from '@mui/material';
import { FileText, Download, Clock } from 'lucide-react';
import axios from 'axios';
import { useQuery } from '@tanstack/react-query';
import { useAssessmentDetail } from '../api/adApi';

const BASE = '/api/plugins/active_directory/assessments';

export const ADReportsPage: React.FC = () => {
  const { assessmentId } = useParams({ strict: false }) as any;
  const id = Number(assessmentId);
  const { data: assessment } = useAssessmentDetail(id);
  const [downloading, setDownloading] = useState<string | null>(null);

  const { data: evidenceLogs, isLoading: loadingLogs } = useQuery({
    queryKey: ['ad-evidence', id],
    queryFn: async () => {
      const { data } = await axios.get(`${BASE}/${id}/evidence/`);
      return data as Array<{ event_type: string; actor: string; details: Record<string, unknown>; timestamp: string }>;
    },
    enabled: !!id,
  });

  const downloadReport = async (format: 'pdf' | 'json') => {
    setDownloading(format);
    try {
      const response = await axios.get(`${BASE}/${id}/report/`, {
        params: { format },
        responseType: 'blob',
      });
      const url = URL.createObjectURL(response.data);
      const a = document.createElement('a');
      a.href = url;
      a.download = `AD_Report_${assessment?.name || id}.${format}`;
      a.click();
      URL.revokeObjectURL(url);
    } catch (err) {
      console.error('Download failed:', err);
    } finally {
      setDownloading(null);
    }
  };

  if (!assessment) return <Box sx={{ p: 4 }}><CircularProgress /></Box>;

  if (!['COMPLETED', 'RUNNING'].includes(assessment.status)) {
    return (
      <Box sx={{ p: 3 }}>
        <Alert severity="info">
          Reports are available once an assessment is running or completed.
        </Alert>
      </Box>
    );
  }

  return (
    <Box sx={{ p: 3 }}>
      <Typography variant="h6" sx={{ fontFamily: 'Orbitron', letterSpacing: 2, mb: 3 }}>
        REPORTS & EVIDENCE
      </Typography>

      {/* Download options */}
      <Paper sx={{ p: 3, mb: 3 }}>
        <Typography variant="subtitle2" sx={{ mb: 2, fontFamily: 'Orbitron', fontSize: '0.75rem', letterSpacing: 1 }}>
          EXPORT REPORTS
        </Typography>
        <Box sx={{ display: 'flex', gap: 2 }}>
          <Button
            variant="contained"
            startIcon={downloading === 'pdf' ? <CircularProgress size={16} /> : <Download size={16} />}
            onClick={() => downloadReport('pdf')}
            disabled={!!downloading}
          >
            Executive Report (PDF)
          </Button>
          <Button
            variant="outlined"
            startIcon={downloading === 'json' ? <CircularProgress size={16} /> : <FileText size={16} />}
            onClick={() => downloadReport('json')}
            disabled={!!downloading}
          >
            Technical Report (JSON)
          </Button>
        </Box>
      </Paper>

      {/* Evidence timeline */}
      <Paper sx={{ p: 3 }}>
        <Typography variant="subtitle2" sx={{ mb: 2, fontFamily: 'Orbitron', fontSize: '0.75rem', letterSpacing: 1 }}>
          ASSESSMENT TIMELINE

          <CircularProgress size={20} />

          <Typography variant="body2" sx={{ color: 'text.secondary' }}>
            No evidence events recorded yet.

                      </Box>

                    secondary={
                      <Typography variant="caption" sx={{ fontFamily: 'monospace', color: 'text.secondary' }}>

                      </Typography>
                    }
                  />
                </ListItem>
                {idx < evidenceLogs.length - 1 && <Divider variant="inset" component="li" />}
              </React.Fragment>
            ))}
          </List>
        )}
      </Paper>
    </Box>
  );
};
```
